Cálculo de precio — siempre en el servidor

// Al agregar/quitar packs o items en cotización:
precioListaUF = SUM(items.precioListaUF)
descuentoPacksUF = SUM(packsAplicados.descuentoAplicadoUF)
precioFinalUF = precioListaUF − descuentoPacksUF − descuentoAprobadoUF
